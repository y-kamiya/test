1 {-# LANGUAGE DeriveDataTypeable #-}
    2 
    3 import System.Environment
    4 import System.Console.CmdArgs
    5 
    6 modeList = ["hoge", "fuga"]
    7 
    8 data Modes = Sample {hello :: String} 
    9             | Hoge {
   10                    hoges :: String
   11               }
   12             | Fuga {
   13                    fugas :: String
   14               }
   15             deriving (Show, Data, Typeable)
   16 
   17 detail = details . lines
   18 
   19 sample :: Modes
   20 sample = Sample{hello = def &= help "World argument" &= opt "world"}
   21 
   22 hoge :: Modes
   23 hoge = Hoge {
   24     hoges = def &= explicit &= name "ho" &= help "aaa"
   25 } &= name "hogehoge"
   26   &= details ["bbbbbbbbbbbb"]
   27 
   28 fuga :: Modes
   29 fuga = Fuga {
   30     fugas = def
   31 }
   32 
   33 samples = cmdArgsMode $ modes [sample, hoge, fuga] 
   34          &= summary "Sample v1"
   35 
   36 main = do
   37     name <- getProgName
   38     args <- getArgs
   39     print name
   40     print args
   41     a <- cmdArgsRun samples
   42     print a
